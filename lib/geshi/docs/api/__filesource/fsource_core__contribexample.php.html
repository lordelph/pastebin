18        } else {

25    if ( isset($_POST['submit']) )
26    {
27        if ( get_magic_quotes_gpc() ) $_POST['source'] = stripslashes($_POST['source']);
28        if ( !strlen(trim($_POST['source'])) )
29        {
30            $_POST['language'] = preg_replace('#[^a-zA-Z0-9\-_]#', '', $_POST['language']);
31            $_POST['source'] = implode('', @file($path . 'geshi/' . $_POST['language'] . '.php'));
32            $_POST['language'] = 'php';
33        }
34        
35        // Here's a free demo of how GeSHi works.
36        
37        // First the initialisation: source code to highlight and the language to use. Make sure
38        // you sanitise correctly if you use $_POST of course - this very script has had a security
39        // advisory against it in the past because of this. Please try not to use this script on a
40        // live site.
41        $geshi =& new GeSHi($_POST['source'], $_POST['language']);
42        
43        // Use the PRE header. This means less output source since we don't have to output &nbsp;
44        // everywhere. Of course it also means you can't set the tab width.
45        $geshi->set_header_type(GESHI_HEADER_PRE);
46        
47        // Enable CSS classes. You can use get_stylesheet() to output a stylesheet for your code. Using
48        // CSS classes results in much less output source.
49        $geshi->enable_classes();
50        
51        // Enable line numbers. We want fancy line numbers, and we want every 5th line number to be fancy
52        $geshi->enable_line_numbers(GESHI_FANCY_LINE_NUMBERS, 5);
53        
54        // Set the style for the PRE around the code. The line numbers are contained within this box (not
55        // XHTML compliant btw, but if you are liberally minded about these things then you'll appreciate
56        // the reduced source output).
57        $geshi->set_overall_style('color: #000066; border: 1px solid #d0d0d0; background-color: #f0f0f0;', true);
58        
59        // Set the style for line numbers. In order to get style for line numbers working, the <li> element
60        // is being styled. This means that the code on the line will also be styled, and most of the time
61        // you don't want this. So the set_code_style reverts styles for the line (by using a <div> on the line).
62        // So the source output looks like this:
63        //
64        // <pre style="[set_overall_style styles]"><ol>
65        // <li style="[set_line_style styles]"><div style="[set_code_style styles]>...</div></li>
66        // ...
67        // </ol></pre>
68        $geshi->set_line_style('font: normal normal 95% \'Courier New\', Courier, monospace; color: #003030;', 'font-weight: bold; color: #006060;', true);
69        $geshi->set_code_style('color: #000020;', 'color: #000020;');
70    
71        // Styles for hyperlinks in the code. GESHI_LINK for default styles, GESHI_HOVER for hover style etc...
72        // note that classes must be enabled for this to work.
73        $geshi->set_link_styles(GESHI_LINK, 'color: #000060;');
74        $geshi->set_link_styles(GESHI_HOVER, 'background-color: #f0f000;');
75    
76        // Use the header/footer functionality. This puts a div with content within the PRE element, so it is
77        // affected by the styles set by set_overall_style. So if the PRE has a border then the header/footer will
78        // appear inside it.
79        $geshi->set_header_content('GeSHi &copy; 2004, Nigel McNie. View source of example.php for example of using GeSHi');
80        $geshi->set_header_content_style('font-family: Verdana, Arial, sans-serif; color: #808080; font-size: 70%; font-weight: bold; background-color: #f0f0ff; border-bottom: 1px solid #d0d0d0; padding: 2px;');
81    
82        // You can use <TIME> and <VERSION> as placeholders
83        $geshi->set_footer_content('Parsed in <TIME> seconds,  using GeSHi <VERSION>');
84        $geshi->set_footer_content_style('font-family: Verdana, Arial, sans-serif; color: #808080; font-size: 70%; font-weight: bold; background-color: #f0f0ff; border-top: 1px solid #d0d0d0; padding: 2px;');
85    }
